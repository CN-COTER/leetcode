#### 快排
def quick_sort(arr, l, r):
    x = arr[l]
    i, j = l, r
    while i < j:
        while arr[j] > x and i < j:
            j -= 1
        if i < j:
            arr[i] = arr[j]

        while arr[i] < x and i < j:
            i += 1
        if i < j:
            arr[j] = arr[i]
    arr[i] = x
    return i

def main(arr, l, r):
    if l < r:
        mid = quick_sort(arr, l, r)
        quick_sort(arr, l, mid-1)
        quick_sort(arr, mid+1, r)
    print(arr)

arr = [1, 2, 3, -1]
main(arr, 0, len(arr)-1)

def minNumber(strs):
    def fast_sort(l, r):
        if l >= r: return
        i, j = l, r
        while i < j:
            while strs[j] >= strs[l] and i < j: j -= 1
            while strs[i] <= strs[l] and i < j: i += 1
            strs[i], strs[j] = strs[j], strs[i]
        strs[i], strs[l] = strs[l], strs[i]
        fast_sort(l, i - 1)
        fast_sort(i + 1, r)
    fast_sort(0, len(strs)-1)

#### sort 与 sorted 区别：
sort 只是应用在 list 上的方法，（就地排序无返回值）。

sorted 是内建函数，可对所有可迭代的对象进行排序操作，（返回新的list）。

 

语法
sorted 语法：

sorted(iterable, key=None, reverse=False)  
参数说明：

iterable -- 可迭代对象。
key -- 主要是用来进行比较的元素，只有一个参数，具体的函数的参数就是取自于可迭代对象中，指定可迭代对象中的一个元素来进行排序。
reverse -- 排序规则，reverse = True 降序 ， reverse = False 升序（默认）。

exec:
a.sort()
b = sorted(a)
